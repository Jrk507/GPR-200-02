void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
	sampler texOne = texture(iChannel0, fragCoord / iResolution.xy);
    sampler texTwo = texture(iChannel1, fragCoord / iResolution.xy);
    fragColor = texOne;
    fragColor = texTwo;
    fragColor = mix(texOne, texTwo, 0.025);
}
